```prisma
// в”Ђв”Ђв”Ђ Authentication & Users в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

model User {
  id            String    @id @default(cuid())
  name          String?
  email         String    @unique
  emailVerified DateTime?
  image         String?
  role          String    @default("viewer") // admin, editor, viewer

model Account {
  id                String  @id @default(cuid())
  userId            String
  type              String

// в”Ђв”Ђв”Ђ Brand Configuration в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

model Brand {
  id                String        @id @default(cuid())
  name              String        @unique
  slug              String        @unique
  colorPrimary      String        // hex
  colorSecondary    String        // hex
  colorAccent       String        // hex
  colorBackground   String        // hex
  voiceDescription  String        @db.Text
  bannedTerms       String[]
  trademarkRules    String        @db.Text
  hashtagStrategy   Json          // { platform: { defaults: [], max_count: N } }
  imagenPrefix      String        @db.Text
  negativePrompt    String        @db.Text
  platformRules     Json          // { platform: { char_limit: N, formatting: "" } }

  brandFonts        Json?         // [{ name: "", fileUrl: "", weight: N }]
  isActive          Boolean       @default(true)

// в”Ђв”Ђв”Ђ Templates в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

model Template {
  id                        String            @id @default(cuid())
  brandId                   String
  brand                     Brand             @relation(fields: [brandId], references: [id], onDelete: Cascade)
  name                      String
  description               String?
  version                   Int               @default(1)
  imagenPromptAddition      String            @db.Text
  imagenNegativeAddition    String            @db.Text
  claudeCaptionInstruction  Json              // { twitter: "", instagram_feed: "", instagram_story: "", linkedin: "" }
  claudeHeadlineInstruction String?           @db.Text
  layoutSpec                Json?             // { headline_zone: {}, logo_placement: {}, gradient_overlay: {}, aspect_ratio_overrides: {} }
  recommendedAssetTypes     String[]
  targetAspectRatios        String[]          // ["1:1", "4:5", "9:16", "16:9", "1.91:1"]
  totalGenerations          Int               @default(0)

  avgRefinementsBeforeApproval Float?
  isActive                  Boolean           @default(true)
  campaigns                 Campaign[]
  refinementLogs            RefinementLog[]
  versions                  TemplateVersion[]
  createdAt                 DateTime          @default(now())
  updatedAt                 DateTime          @updatedAt

  @@unique([brandId, name])
}

model TemplateVersion {
  id             String   @id @default(cuid())
  templateId     String
  template       Template @relation(fields: [templateId], references: [id], onDelete: Cascade)

  changesApplied Json?    // diff from previous version
  appliedBy      String?
  createdAt      DateTime @default(now())

// в”Ђв”Ђв”Ђ Asset Library в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

model Asset {
  id          String   @id @default(cuid())
  brandId     String
  brand       Brand    @relation(fields: [brandId], references: [id], onDelete: Cascade)
  type        String   // "visual" or "context"
  subType     String?  // character_art, gameplay_screenshot, logo, background, partnership_graphic
  status      String   @default("uploaded") // uploaded, approved, archived
  name        String
  description String?  @db.Text
  tags        String[]
  fileUrl     String?  // for visual assets вЂ” storage URL
  thumbnailUrl String? // for visual assets вЂ” generated thumbnail
  content     String?  @db.Text // for context/brief assets вЂ” the text content
  mimeType    String?
  fileSizeBytes Int?
  uploadedBy  String
  approvedBy  String?
  approvedAt  DateTime?
  usageCount  Int      @default(0)
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  @@index([brandId, status])
  @@index([brandId, type])
}

// в”Ђв”Ђв”Ђ Hashtag Library в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

  @@index([brandId, category])

// в”Ђв”Ђв”Ђ Campaigns в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

model Campaign {
  id                String            @id @default(cuid())
  brandId           String
  brand             Brand             @relation(fields: [brandId], references: [id])
  templateId        String?
  template          Template?         @relation(fields: [templateId], references: [id])
  name              String
  status            String            @default("draft") // draft, generating, pending_review, approved, scheduled, published, rejected
  briefText         String?           @db.Text
  referenceUrl      String?
  referenceAnalysis Json?             // Gemini Flash structured analysis output
  referenceScreenshots Json?          // URLs to stored screenshot images
  selectedAssets    Json              // [{ assetId: "", role: "primary"|"supporting", order: N }]
  selectedHashtagSets Json?           // [hashtagSetId, ...]

  estimatedCost     Float?

  refinementLogs    RefinementLog[]
  calendarEntries   CalendarEntry[]
  createdBy         String
  reviewedBy        String?
  reviewComment     String?           @db.Text
  reviewedAt        DateTime?
  createdAt         DateTime          @default(now())
  updatedAt         DateTime          @updatedAt

  @@index([brandId, status])
  @@index([templateId, createdAt])
}

model CampaignVariant {
  id                String         @id @default(cuid())
  campaignId        String

  compositedImageUrl String?       // final image with text + logo + gradient overlays
  videoUrl          String?
  audioSource       String?        // veo_native, elevenlabs_music, elevenlabs_sfx, uploaded, muted
  audioUrl          String?        // ElevenLabs generated or uploaded audio file
  mergedVideoUrl    String?        // video + audio merged via ffmpeg
  caption           String         @db.Text
  originalCaption   String?        @db.Text
  headlineText      String?
  originalHeadline  String?
  status            String         @default("generated") // generated, approved, rejected, scheduled, published, failed
  calendarEntries   CalendarEntry[]
  createdAt         DateTime       @default(now())
  updatedAt         DateTime       @updatedAt

  @@index([campaignId])
}

// в”Ђв”Ђв”Ђ Calendar & Scheduling в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

// в”Ђв”Ђв”Ђ Social Accounts в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

  createdAt       DateTime        @default(now())
  updatedAt       DateTime        @updatedAt

// в”Ђв”Ђв”Ђ Refinement Tracking в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ

// в”Ђв”Ђв”Ђ Cost Tracking в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
```

## 2. API Route Specifications

### 2.1 Authentication

```
POST /api/auth/[...nextauth]     вЂ” NextAuth.js handler (Google OAuth)
GET  /api/auth/session           вЂ” Get current session
```

**NextAuth Configuration:**
- Provider: Google OAuth 2.0
- Callbacks: `signIn` вЂ” check if email is in allowed list or domain is @sparqgames.com
- Callbacks: `session` вЂ” attach user role from database
- Pages: custom sign-in page at `/login`

### 2.2 Brands


### 2.6 Campaigns

```

```

**POST /api/campaigns/[id]/generate вЂ” Request:**
```json
{
  "templateId": "tmpl_id",
  "selectedAssets": [
    { "assetId": "asset_1", "role": "primary" },
    { "assetId": "asset_2", "role": "supporting" }
  ],
  "briefText": "Ohio State Buckeyes partnership announcement...",
  "referenceUrl": "https://example.com/campaign",
  "selectedHashtagSets": ["set_1", "set_2"],
  "includeVideo": false,
  "audioPreference": "veo_native"
}
```

**Response:** `202 Accepted` вЂ” returns campaign ID. Client polls for status updates via:
```
GET /api/campaigns/[id]/status вЂ” Returns generation progress per variant
```

**POST /api/campaigns/[id]/refine вЂ” Request:**
```json
{
  "variantId": "variant_id",
  "refinementPrompt": "darker background, more dramatic lighting",
  "refineAll": false
}
```
This re-runs Imagen for the specified variant with the refinement appended to the prompt, then re-composites.

**POST /api/campaigns/[id]/recomposite вЂ” Request:**
```json
{
  "variantId": "variant_id",
  "newHeadlineText": "OHIO STATE ENTERS THE GAME"
}
```
This re-runs compositing ONLY using the existing raw Imagen image with new headline text. No Imagen API call. Near-instant.

**POST /api/campaigns/[id]/remix вЂ” Request:**
```json
{
  "newTemplateId": "tmpl_different"
}
```
Creates a new campaign with same brand, assets, and brief but different template. Returns the new campaign ID.

**GET /api/campaigns/[id]/duplicate-check вЂ” Response:**
```json
{
  "hasDuplicate": true,
  "similarCampaign": {
    "id": "campaign_xyz",
    "name": "Character Reveal вЂ” Ohio State",
    "createdAt": "2026-03-12T...",
    "templateName": "Character Reveal"
  }
}
```
Logic: `SELECT * FROM Campaign WHERE templateId = :templateId AND selectedAssets->primary = :primaryAssetId AND createdAt > NOW() - INTERVAL '30 days' AND id != :currentCampaignId`

```json
{

  "prompt": "8-second high-energy esports intro, electronic, building tension",
```

Modes: `replace` (strip original audio, use new), `mix` (blend both), `mute` (strip all audio)

### 2.8 Reference URL


## 6. Compositing Pipeline

### Architecture

```
Raw Imagen Image + Layout Spec + Headline Text + Logo в†’ Composited Image
```

### Implementation with Sharp + node-canvas

```typescript
// lib/compositing.ts
import sharp from "sharp";
import { createCanvas, registerFont, loadImage } from "canvas";

interface CompositingInput {
  rawImageBuffer: Buffer;
  layoutSpec: LayoutSpec;
  headlineText: string;
  logoBuffer: Buffer | null;
  aspectRatio: string;
  outputWidth: number;  // target pixel width
  outputHeight: number; // target pixel height
}

async function compositeImage(input: CompositingInput): Promise<Buffer> {
  const { rawImageBuffer, layoutSpec, headlineText, logoBuffer, aspectRatio } = input;

  // 1. Resolve aspect-ratio-specific overrides
  const headlineZone = layoutSpec.aspect_ratio_overrides?.[aspectRatio]?.headline_zone
    ?? layoutSpec.headline_zone;
  const logoPlacement = layoutSpec.aspect_ratio_overrides?.[aspectRatio]?.logo_placement
    ?? layoutSpec.logo_placement;
  const gradient = layoutSpec.aspect_ratio_overrides?.[aspectRatio]?.gradient_overlay
    ?? layoutSpec.gradient_overlay;

  // 2. Load and resize raw image to target dimensions
  const baseImage = sharp(rawImageBuffer).resize(input.outputWidth, input.outputHeight);
  const baseBuffer = await baseImage.toBuffer();
  const { width, height } = await sharp(baseBuffer).metadata();

  // 3. Create canvas for overlays
  const canvas = createCanvas(width!, height!);
  const ctx = canvas.getContext("2d");
  const base = await loadImage(baseBuffer);
  ctx.drawImage(base, 0, 0);

  // 4. Apply gradient overlay
  if (gradient) {
    applyGradient(ctx, width!, height!, gradient);
  }

  // 5. Render headline text
  if (headlineText && headlineZone) {
    renderHeadline(ctx, width!, height!, headlineText, headlineZone);
  }

  // 6. Place logo
  if (logoBuffer && logoPlacement) {
    await placeLogo(ctx, width!, height!, logoBuffer, logoPlacement);
  }

  // 7. Export as PNG
  return canvas.toBuffer("image/png");
}
```

### Font Loading

```typescript
// Load brand fonts at compositing time
function loadBrandFonts(brandFonts: BrandFont[]): void {
```

```typescript
// lib/ffmpeg.ts
import { execFile } from "child_process";
import { promisify } from "util";
const exec = promisify(execFile);

async function mergeAudioVideo(
  videoPath: string,
  audioPath: string,
  outputPath: string,
  mode: "replace" | "mix" | "mute"
): Promise<void> {
  switch (mode) {
    case "replace":
      await exec("ffmpeg", ["-i", videoPath, "-i", audioPath,
        "-c:v", "copy", "-c:a", "aac", "-map", "0:v:0", "-map", "1:a:0",
        "-y", outputPath]);
      break;
    case "mix":
      await exec("ffmpeg", ["-i", videoPath, "-i", audioPath,
        "-filter_complex", "[0:a][1:a]amix=inputs=2:duration=first:dropout_transition=2[aout]",
        "-map", "0:v", "-map", "[aout]", "-c:v", "copy", "-c:a", "aac",
        "-y", outputPath]);
      break;
    case "mute":
      await exec("ffmpeg", ["-i", videoPath, "-an", "-c:v", "copy", "-y", outputPath]);
      break;
  }
}
```

## 8. Error Handling Patterns

### Per-Model Error Handling

| Model | Common Errors | Handling |
|-------|--------------|---------|
| **Claude** | Rate limit (429), context too long, content filter | Retry with exponential backoff (max 3). If content filter, simplify prompt and retry once. Log error. |
| **Imagen** | Rate limit (429), safety filter, invalid aspect ratio | Retry with backoff (max 3). If safety filter, modify prompt (remove potentially flagged terms) and retry. If invalid aspect ratio, use nearest supported. |
| **Veo** | Timeout (generation >90s), safety filter, quota exceeded | Longer timeout (120s). Retry once on timeout. If safety filter, modify prompt. If quota, show "Video generation unavailable вЂ” daily limit reached." |
| **Gemini Flash** | Rate limit, invalid image format | Retry with backoff. Validate image format before sending. |
| **ElevenLabs** | Credit limit, invalid prompt, rate limit | Check credits before generation. Retry on rate limit. If credit limit, show "Audio generation unavailable вЂ” credits exhausted." |
| **ScreenshotOne** | URL unreachable, timeout, quota | Show specific error message. Offer "upload screenshot manually" fallback. |
| **Compositing** | Font not found, invalid layout spec, Sharp error | Fall back to system font if brand font unavailable. Validate layout spec before compositing. Log and skip overlay if individual element fails. |
| **ffmpeg** | File not found, invalid format, process crash | Validate inputs exist before calling. Timeout after 30s. Return video without audio merge on failure. |

### Error Response Format

```json
{
  "error": {
    "code": "IMAGEN_SAFETY_FILTER",
    "message": "Image generation was blocked by safety filters. Try adjusting the prompt.",
    "retryable": true,
    "variantId": "variant_id",
    "suggestions": ["Remove references to violence", "Simplify the scene description"]
  }
}
```

### Variant-Level Isolation

Critical design: if one variant fails, others are unaffected. Each variant in the Campaign Studio has its own status indicator and retry button. A failed Twitter variant does not block the Instagram variant.

## 9. Context Assembly Algorithm (Pseudocode)

```
FUNCTION assembleAndGenerate(campaignId, inputs):

  // в”Ђв”Ђ STEP 1: Load all context from database в”Ђв”Ђ
  brand = db.Brand.findById(inputs.brandId)
  template = db.Template.findById(inputs.templateId)
  assets = db.Asset.findByIds(inputs.selectedAssets.map(a => a.assetId))
  hashtagSets = db.HashtagSet.findByIds(inputs.selectedHashtagSets)
  primaryAsset = assets.find(a => a.role == "primary")
  supportingAssets = assets.filter(a => a.role == "supporting")

  // в”Ђв”Ђ STEP 2: Reference URL analysis (if provided, must complete first) в”Ђв”Ђ
  referenceAnalysis = null
  referenceScreenshots = []
  IF inputs.referenceUrl:
    screenshots = await screenshotOne.capture(inputs.referenceUrl)
    referenceAnalysis = await geminiFlash.analyze(screenshots)
    referenceScreenshots = screenshots
    store referenceAnalysis and screenshots in campaign record

  // в”Ђв”Ђ STEP 3: Duplicate check в”Ђв”Ђ
  duplicate = db.Campaign.findFirst({
    WHERE: templateId == inputs.templateId
      AND selectedAssets.primary == primaryAsset.id
      AND createdAt > NOW() - 30 days
      AND id != campaignId
  })
  IF duplicate: emit duplicateWarning to client

  // в”Ђв”Ђ STEP 4: Assemble Claude prompt package в”Ђв”Ђ
  claudeSystemPrompt = buildClaudeSystem(brand)
    // Includes: voice, tone, banned terms, trademark rules, platform rules
    // Includes: hashtag library sets as reference

  claudeUserMessage = buildClaudeUser(template, inputs, referenceAnalysis, primaryAsset)
    // Includes: template caption instruction (per platform)
    // Includes: template headline instruction
    // Includes: brief text
    // Includes: reference analysis (if available)
    // Includes: primary asset image (base64)
    // Includes: instruction to return JSON with platform-keyed captions + headlines

  // в”Ђв”Ђ STEP 5: Assemble Imagen prompt packages (one per aspect ratio) в”Ђв”Ђ
  FOR EACH aspectRatio IN template.targetAspectRatios:
    imagenPrompt = concatenate(
      brand.imagenPrefix,
      template.imagenPromptAddition,
      inputs.briefText,
      referenceAnalysis?.forImagen(),  // visual mood + composition techniques
    )
    imagenNegative = concatenate(
      brand.negativePrompt,
      template.imagenNegativeAddition,
      "no text baked into image, no words or letters in the image"
    )
    imagenReferenceImages = [primaryAsset.file, ...supportingAssets.map(a => a.file)]

    CREATE imagenRequest { prompt, negativePrompt, aspectRatio, referenceImages }

  // в”Ђв”Ђ STEP 6: Execute API calls в”Ђв”Ђ
  // Claude and Imagen run in parallel
  // Claude returns first (~1-2s) в†’ captions + headlines populate in UI
  // Imagen returns per variant (~2.5s each) в†’ raw images populate

  claudeResult = ASYNC claude.generate(claudeSystemPrompt, claudeUserMessage)
  imagenResults = ASYNC_ALL imagenRequests.map(req => imagen.generate(req))

  // в”Ђв”Ђ STEP 7: Process Claude results в”Ђв”Ђ
  WHEN claudeResult resolves:
    parsedCaptions = JSON.parse(claudeResult)
    FOR EACH platform:
      UPDATE variant SET caption = parsedCaptions[platform].caption,
                         headlineText = parsedCaptions[platform].headline,
                         originalCaption = parsedCaptions[platform].caption,
                         originalHeadline = parsedCaptions[platform].headline
      EMIT captionReady(variantId) to client

  // в”Ђв”Ђ STEP 8: Process Imagen results + compositing в”Ђв”Ђ
  WHEN EACH imagenResult resolves:
    store rawImage in storage
    UPDATE variant SET rawImageUrl = rawImage.url

    // Compositing step (near-instant, <500ms)
    compositedImage = compositeImage({
      rawImage,
      layoutSpec: template.layoutSpec,
      headlineText: variant.headlineText,  // from Claude
      logo: brand.logoFile,
      aspectRatio: variant.aspectRatio
    })
    store compositedImage in storage
    UPDATE variant SET compositedImageUrl = compositedImage.url
    EMIT variantReady(variantId) to client

  // в”Ђв”Ђ STEP 9: Video generation (if requested, runs last) в”Ђв”Ђ
  IF inputs.includeVideo:
    veoResult = ASYNC veo.generate({
      prompt: assembledVeoPrompt,
      aspectRatios: ["16:9", "9:16"],
      referenceImages: [primaryAsset.file]
    })
    // Veo takes 15-60s вЂ” show progress indicator
    WHEN veoResult resolves:
      store video
      UPDATE variant SET videoUrl, audioSource = "veo_native"
      EMIT videoReady(variantId) to client

  // в”Ђв”Ђ STEP 10: Audio generation (if custom audio requested) в”Ђв”Ђ
  IF inputs.audioPreference != "veo_native":
    audioResult = await generateAudio(inputs.audioPreference, inputs.audioPrompt)
    mergedVideo = await mergeAudioVideo(variant.videoUrl, audioResult, mode)
    UPDATE variant SET audioUrl, mergedVideoUrl, audioSource
    EMIT audioReady(variantId) to client

  // в”Ђв”Ђ STEP 11: Log costs в”Ђв”Ђ
  logCosts(campaignId, allApiCallCosts)
  UPDATE campaign SET estimatedCost = sum(allApiCallCosts)
```

## 10. Scheduled Publishing (Phase 5)

### v1 Approach: PostgreSQL Polling

Instead of BullMQ + Redis, use a simpler polling approach:

```typescript
// app/api/publish/process-queue/route.ts
// Called by a cron job or setInterval every 60 seconds

export async function POST() {
  const duePosts = await prisma.calendarEntry.findMany({
    where: {
      publishStatus: "scheduled",
      scheduledAt: { lte: new Date() },
    },
    include: { variant: true, socialAccount: true },
    take: 10, // process in batches
  });

  for (const entry of duePosts) {
    await prisma.calendarEntry.update({
      where: { id: entry.id },
      data: { publishStatus: "publishing" },
    });

    try {
      await publishToplatform(entry);
      await prisma.calendarEntry.update({
        where: { id: entry.id },
        data: { publishStatus: "published", publishedAt: new Date() },
      });
    } catch (error) {
      await prisma.calendarEntry.update({
        where: { id: entry.id },
        data: {
          publishStatus: entry.retryCount < 3 ? "scheduled" : "failed",
          publishError: error.message,
          retryCount: { increment: 1 },
        },
      });
    }
  }
}
```

Set up a Replit cron or `setInterval` in a keep-alive route to call this endpoint every minute.

## 11. Security Considerations

- All API keys stored in Replit Secrets (environment variables), never in code
- OAuth tokens encrypted at rest in database (use Prisma middleware or field-level encryption)
- All API routes check authentication via `getServerSession()`
- Role-based middleware validates user permission before destructive operations
- File uploads validated: check MIME type, enforce size limits (10MB images, 50MB video, 5MB audio, 2MB fonts)
- Rate limiting on generation endpoints (prevent accidental cost spikes)
- CSRF protection via NextAuth.js built-in mechanisms
- Input sanitization on all user-provided text (Zod validation on every API route)
